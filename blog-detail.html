3. State Management with TypeScript

Managing state in a TypeScript React application requires careful consideration of types. Here's how we can use the useState hook with proper typing. 

interface Todo {
  id: number;
  text: string;
  completed: boolean;
}

const [todos, setTodos] = useState([]);

const addTodo = (text: string) => {
  const newTodo: Todo = {
    id: Date.now(),
    text,
    completed: false
  };
  setTodos([...todos, newTodo]);
};

4. Custom Hooks with TypeScript

Custom hooks are a great way to reuse stateful logic. Let's create a custom hook for handling form inputs. 

interface UseFormInput {
  value: string;
  onChange: (e: React.ChangeEvent) => void;
  reset: () => void;
}

const useFormInput = (initialValue: string): UseFormInput => {
  const [value, setValue] = useState(initialValue);

  const handleChange = (e: React.ChangeEvent) => {
    setValue(e.target.value);
  };

  const reset = () => {
    setValue(initialValue);
  };

  return {
    value,
    onChange: handleChange,
    reset
  };
};
